\section{Blocos}

\subsection{Teorema}

% --- Bloco de Teorema ---

\begin{frame}{Bloco de Teorema}
    
    % Construído com tons de rosa (fixo)
    % Função: Utilizado para enunciar teoremas.
    \begin{theorem}
        Se uma função $f$ for contínua no intervalo fechado $[a, b]$, então a função $g$ definida por:
        $$g(x) = \int_{a}^{x} f(t) \, dt$$
        é contínua em $[a, b]$, derivável em $(a, b)$, e sua derivada é:
        $$g'(x) = f(x)$$
    \end{theorem}

\end{frame}

\subsection{Prova}

\begin{frame}{Bloco de Prova}

% --- Bloco de Prova ---

    % Construído com tons de cinza (fixo)
    % Função: Utilizado para apresentar a demonstração matemática de um teorema.
    \begin{proof}
        Pela definição formal de derivada, avaliamos o limite da taxa de variação:
        $$g'(x) = \lim_{h \to 0} \frac{1}{h} \int_{x}^{x+h} f(t) \, dt$$
        Como $f$ é contínua, quando $h \to 0$, o valor médio da integral aproxima-se de $f(x)$, finalizando a demonstração.
    \end{proof}

\end{frame}

\subsection{Exemplo}

\begin{frame}{Bloco de Exemplo}

% --- Bloco de Exemplo --- 

    % Construído com tons de verde (fixo)
    % Função: Ideal para ilustrar a teoria abordada anteriormente com casos de uso práticos, numéricos ou aplicados.
    \begin{exampleblock}{Otimização com Gradiente Descendente}
        Na regressão linear, ajustamos os pesos iterativamente para minimizar a função de custo $J(w)$. O passo de atualização é dado por:
        $$w_{i} \leftarrow w_{i} - \alpha \frac{\partial J}{\partial w_{i}}$$
        Com uma taxa de aprendizado adequada ($\alpha = 0.01$), o algoritmo converge suavemente para o mínimo global.
    \end{exampleblock}
    
\end{frame}

\subsection{Alerta}

\begin{frame}{Bloco de Alerta}

% --- Bloco de Alerta --- 

    % Construído com tons de laranja (fixo)
    % Função: Usado para chamar a atenção para avisos críticos, exceções ou erros comuns.
    \begin{alertblock}{Risco de \textit{Overfitting} (Sobreajuste)}
        Ao treinar modelos complexos com conjuntos de dados pequenos, a rede pode 
        memorizar os exemplos de treino em vez de aprender a generalizar. Sempre utilize 
        validação cruzada e reserve um conjunto de teste para evitar essa armadilha 
        metodológica.
    \end{alertblock}
    
\end{frame}

\subsection{Código}

% Todo slide que contém código (como lstlisting ou verbatim) exige
% obrigatoriamente a opção [fragile] no \begin{frame}. A opção
% [allowframebreaks] é recomendada para códigos longos, pois 
% divide automaticamente o slide em várias telas se o código não couber.
\begin{frame}[fragile, allowframebreaks]{Código - Parte}
    Estrutura de uma função em Python:

% Inicia o ambiente de código, com os parâmetros:
% > language=Python: Aplica as cores e o realce de sintaxe da linguagem Python.
% > showstringspaces=false: Remove o símbolo "u" que marca os espaços dentro de textos (strings).
\begin{lstlisting}[language=Python, showstringspaces=false]
import random
import string

def gerar_senha(tamanho = 12):
    '''Gera uma senha segura com letras, números e símbolos.'''
    if tamanho < 4:
        return 'Erro: Tamanho mínimo de 4 caracteres.'
        
    caracteres = string.ascii_letters + string.digits + string.punctuation
    
    # Garante a presença de diferentes tipos de caracteres
    senha = [
        random.choice(string.ascii_lowercase),
        random.choice(string.ascii_uppercase),
        random.choice(string.digits),
        random.choice(string.punctuation)
    ]
    
    # Preenche o restante do tamanho desejado
    senha += random.choices(caracteres, k = tamanho - 4)
    
    # Embaralha para não seguir um padrão previsível
    random.shuffle(senha)
    
    return ''.join(senha)

# Execução do código
nova_senha = gerar_senha(16)
print(f'Senha gerada: {nova_senha}')
\end{lstlisting}
\end{frame}
