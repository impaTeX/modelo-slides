\section{Blocos}

\subsection{Teorema}

\begin{frame}{Bloco de Teorema}

% --- Bloco de Teorema ---

    % Construído com tons de rosa (fixo)
    \begin{theorem}[(Infinidade dos Primos).]
        O conjunto dos números primos é infinito. Ou seja, não importa quão grande seja o 
        número primo que você encontre, sempre existirá um primo ainda maior, garantindo 
        que a sequência não tem fim.
    \end{theorem}

\end{frame}

\subsection{Prova}

\begin{frame}{Bloco de Prova}

% --- Bloco de Prova ---

    % Construído com tons de cinza (fixo)
    \begin{proof}
        Suponha que existam apenas finitamente muitos primos $p_1, p_2, \dots, p_n$. 
    
        Considere o número $P = (p_1 \times p_2 \times \dots \times p_n) + 1$.
    
        \begin{itemize}
            \item $P$ não é divisível por nenhum dos primos da nossa lista, pois o resto da divisão por qualquer $p_i$ será sempre $1$.
            \item Logo, ou $P$ é ele próprio um novo primo, ou possui um fator primo que não está na lista original.
        \end{itemize}
    
        Em ambos os casos, encontramos um primo que não estava no conjunto finito original, gerando uma contradição. Portanto, o conjunto dos primos é infinito.
  \end{proof}

\end{frame}

\subsection{Exemplo}

\begin{frame}{Bloco de Exemplo}

% --- Bloco de Exemplo --- 

    % Construído com tons de verde (fixo)
    \begin{exampleblock}{Otimização com Gradiente Descendente}
        Na regressão linear, ajustamos os pesos iterativamente para minimizar a função de custo $J(w)$. O passo de atualização é dado por:
        $$w_{i} \leftarrow w_{i} - \alpha \frac{\partial J}{\partial w_{i}}$$
        Com uma taxa de aprendizado adequada ($\alpha = 0.01$), o algoritmo converge suavemente para o mínimo global.
    \end{exampleblock}
    
\end{frame}

\subsection{Ambiente matemático}

\begin{frame}{Ambiente matemático (dentro de bloco)}
    
    % Testando o bloco de Teorema
    \begin{theorem}[Parte 1]
        Se uma função $f$ for contínua no intervalo fechado $[a, b]$, então a função $g$ definida por:
        $$g(x) = \int_{a}^{x} f(t) \, dt$$
        é contínua em $[a, b]$, derivável em $(a, b)$, e sua derivada é:
        $$g'(x) = f(x)$$
    \end{theorem}

\end{frame}

\begin{frame}[allowframebreaks]{Ambiente matemático (dentro de bloco)}

    % Testando o bloco de Prova (Proof) com a nota de rodapé [frame]
    \begin{proof}
        Pela definição formal de derivada, avaliamos o limite da taxa de variação\footnote[frame]{Lembrando que aplicamos o Teorema do Valor Médio para Integrais aqui.}:
        $$g'(x) = \lim_{h \to 0} \frac{1}{h} \int_{x}^{x+h} f(t) \, dt$$
        Como $f$ é contínua, quando $h \to 0$, o valor médio da integral aproxima-se de $f(x)$, finalizando a demonstração.
    \end{proof}

\end{frame}

\begin{frame}{Ambiente Matemático (fora de bloco)}

    Vamos considerar a equação quadrática:
    \begin{equation}
        ax^2 + bx + c = 0
    \end{equation}
    A solução para esta equação é dada pela fórmula de Bhaskara:
    \begin{equation}
        x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
    \end{equation}

\end{frame}
