\subsection{Teorema e Prova}

\begin{frame}{Teorema e prova}

\begin{theorem}[(Infinidade dos Primos).]
    Existem infinitos números primos.
  \end{theorem}

\end{frame}

\begin{frame}{Prova do Teorema}

  \begin{proof}
    Suponha que existam apenas finitamente muitos primos $p_1, p_2, \dots, p_n$. 
    
    Considere o número $P = (p_1 \times p_2 \times \dots \times p_n) + 1$.
    
    \begin{itemize}
        \item $P$ não é divisível por nenhum dos primos da nossa lista, pois o resto da divisão por qualquer $p_i$ será sempre $1$.
        \item Logo, ou $P$ é ele próprio um novo primo, ou possui um fator primo que não está na lista original.
    \end{itemize}
    
    Em ambos os casos, encontramos um primo que não estava no conjunto finito original, gerando uma contradição. Portanto, o conjunto dos primos é infinito.
  \end{proof}

\end{frame}

\begin{frame}{O Teorema Fundamental do Cálculo}
    
    % Testando o bloco de Teorema que configuramos no seu innertheme
    \begin{theorem}[Parte 1]
        Se uma função $f$ for contínua no intervalo fechado $[a, b]$, então a função $g$ definida por:
        $$g(x) = \int_{a}^{x} f(t) \, dt$$
        é contínua em $[a, b]$, derivável em $(a, b)$, e sua derivada é:
        $$g'(x) = f(x)$$
    \end{theorem}

\end{frame}

\begin{frame}{O Teorema Fundamental do Cálculo - Prova}

    % Testando o bloco de Prova (Proof) com a nota de rodapé mágica [frame]
    \begin{proof}
        Pela definição formal de derivada, avaliamos o limite da taxa de variação\footnote[frame]{Lembrando que aplicamos o Teorema do Valor Médio para Integrais aqui.}:
        $$g'(x) = \lim_{h \to 0} \frac{1}{h} \int_{x}^{x+h} f(t) \, dt$$
        Como $f$ é contínua, quando $h \to 0$, o valor médio da integral aproxima-se de $f(x)$, finalizando a demonstração.
    \end{proof}

\end{frame}
